Examples
--------

.. code-block:: yaml

    - name: create database in rethinkdb
      community.missing_collection.rethinkdb_db:
        host: "localhost"
        port: 28015
        user: 'admin'
        password: ''
        state: present
        database: "test1"

    - name: delete database in rethinkdb
      community.missing_collection.rethinkdb_db:
        host: "localhost"
        port: 28015
        user: 'admin'
        password: ''
        state: absent
        database: "test1"
