Examples
--------

.. code-block:: yaml

    - name: create table in rethinkdb
      community.missing_collection.rethinkdb_table:
        host: "localhost"
        port: 28015
        user: 'admin'
        password: ''
        state: present
        database: 'database1'
        table: 'table1'

    - name: delete table in rethinkdb
      community.missing_collection.rethinkdb_table:
        host: "localhost"
        port: 28015
        user: 'admin'
        password: ''
        state: absent
        database: 'database1'
        table: 'table1'
