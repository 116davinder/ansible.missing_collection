Examples
--------

.. code-block:: yaml

    - name: create ssl test
      community.missing_collection.statuscake_ssl:
        api_key: 'Ohxxxxxxxxxxxxxxxxpi'
        command: 'create'
        website_url: 'https://www.google.com'
        alert_at_csv: "30,7,1"
      register: __id

    - name: update ssl test check rate
      community.missing_collection.statuscake_ssl:
        api_key: 'Ohxxxxxxxxxxxxxxxxpi'
        command: 'update'
        id: '{{ __id.id }}'
        check_rate: 86400

    - name: delete ssl test
      community.missing_collection.statuscake_ssl:
        api_key: 'Ohxxxxxxxxxxxxxxxxpi'
        command: 'delete'
        id: '{{ __id.id }}'
