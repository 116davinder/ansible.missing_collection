Examples
--------

.. code-block:: yaml

    - name: list all orientdb databases
      community.missing_collection.orientdb_db_info:
        host: 'localhost'
        port: 2424
        user: 'root'
        password: 'root'
        db_list: true

    - name: get database size in orientdb
      community.missing_collection.orientdb_db_info:
        host: 'localhost'
        port: 2424
        user: 'root'
        password: 'root'
        db_size: 'true'
        database: 'test2'

    - name: check database existence in orientdb
      community.missing_collection.orientdb_db_info:
        host: 'localhost'
        port: 2424
        user: 'root'
        password: 'root'
        db_exists: true
        database: 'test2'
        type: 'plocal'

    - name: get total number of records in orientdb database
      community.missing_collection.orientdb_db_info:
        host: 'localhost'
        port: 2424
        user: 'root'
        password: 'root'
        db_count_records: true
        database: 'test2'

    - name: get list of database with default credentials
      community.missing_collection.orientdb_db_info:
        host: 'localhost'
        port: 2424
        user: 'root'
        password: 'root'
        check_default_credentials: true
