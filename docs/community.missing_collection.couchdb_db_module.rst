Examples
--------

.. code-block:: yaml

    - name: create database in couchdb
      community.missing_collection.couchdb_db:
        scheme: 'http'
        host: 'localhost'
        port: '5984'
        user: 'admin'
        password: 'password'
        state: 'present'
        database: 'test1'

    - name: delete database in couchdb
      community.missing_collection.couchdb_db:
        scheme: 'http'
        host: 'localhost'
        port: '5984'
        user: 'admin'
        password: 'password'
        state: 'absent'
        database: 'test1'
