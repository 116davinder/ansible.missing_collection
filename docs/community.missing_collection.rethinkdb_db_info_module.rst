Examples
--------

.. code-block:: yaml

    - name: list all database in rethinkdb cluster
      community.missing_collection.rethinkdb_db_info:
        host: 'localhost'
        port: 28015
        user: 'admin'
        password: ''
        list_databases: true

    - name: get information about given rethinkdb database
      community.missing_collection.rethinkdb_db_info:
        host: 'localhost'
        port: 28015
        user: 'admin'
        password: ''
        list_database_config: true
        database: 'database1'

    - name: get list of tables of given rethinkdb database
      community.missing_collection.rethinkdb_db_info:
        host: 'localhost'
        port: 28015
        user: 'admin'
        password: ''
        list_tables: true
        database: 'database1'
