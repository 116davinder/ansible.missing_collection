Examples
--------

.. code-block:: yaml

    - name: create lease in etcd for 1000 seconds
      community.missing_collection.etcd3_lease:
        host: "localhost"
        port: 2379
        ttl: 1000
        state: "present"
      register: __

    - name: revoke lease in etcd
      community.missing_collection.etcd3_lease:
        host: "localhost"
        port: 2379
        id: "{{ __.id }}"
        state: "absent"
